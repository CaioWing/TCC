% USPSC-Simbolos.tex
\begin{simbolos}
  \item[$Q$] Matriz \textit{Query}
  \item[$K$] Matriz \textit{Key}
  \item[$V$] Matriz \textit{Value}
  \item[$K^T$] Transposta da matriz $K$
  \item[$d_k$] Dimensionalidade dos vetores de chave
  \item[$N$] Número de \textit{patches} (tokens) da entrada
  \item[$O(N^2)$] Complexidade computacional quadrática em função de $N$
\end{simbolos}
